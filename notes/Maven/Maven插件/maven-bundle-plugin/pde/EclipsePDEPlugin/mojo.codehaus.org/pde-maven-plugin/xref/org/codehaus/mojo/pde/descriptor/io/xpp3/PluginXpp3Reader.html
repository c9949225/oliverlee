301     /**
302      * Method parseDescriptor
303      * 
304      * @param tagName
305      * @param encoding
306      * @param strict
307      * @param parser
308      */
309     private Descriptor parseDescriptor(String tagName, XmlPullParser parser, boolean strict, String encoding)
310         throws IOException, XmlPullParserException
311     {
312         Descriptor descriptor = new Descriptor();
313         descriptor.setModelEncoding( encoding );
314         descriptor.setId( getTrimmedValue( parser.getAttributeValue( "", "id" )) );
315         descriptor.setName( getTrimmedValue( parser.getAttributeValue( "", "name" )) );
316         descriptor.setVersion( getTrimmedValue( parser.getAttributeValue( "", "version" )) );
317         java.util.Set parsed = new java.util.HashSet();
318         while ( parser.nextTag() == XmlPullParser.START_TAG )
319         {
320             if ( strict )
321             {
322                 throw new XmlPullParserException( "Unrecognised tag: '" + parser.getName() + "'", parser, null);
323             }
324         }
325         return descriptor;
326     } //-- Descriptor parseDescriptor(String, XmlPullParser, boolean, String)

328     /**
329      * Method parsePluginDescriptor
330      * 
331      * @param tagName
332      * @param encoding
333      * @param strict
334      * @param parser
335      */
336     private PluginDescriptor parsePluginDescriptor(String tagName, XmlPullParser parser, boolean strict, String encoding)
337         throws IOException, XmlPullParserException
338     {
339         PluginDescriptor pluginDescriptor = new PluginDescriptor();
340         pluginDescriptor.setModelEncoding( encoding );
341         pluginDescriptor.setId( getTrimmedValue( parser.getAttributeValue( "", "id" )) );
342         pluginDescriptor.setName( getTrimmedValue( parser.getAttributeValue( "", "name" )) );
343         pluginDescriptor.setVersion( getTrimmedValue( parser.getAttributeValue( "", "version" )) );
344         java.util.Set parsed = new java.util.HashSet();
345         int eventType = parser.getEventType();
346         boolean foundRoot = false;
347         while ( eventType != XmlPullParser.END_DOCUMENT )
348         {
349             if ( eventType == XmlPullParser.START_TAG )
350             {
351                 if ( parser.getName().equals( tagName ) )
352                 {
353                     foundRoot = true;
354                 }
355                 else
356                 {
357                     if ( foundRoot )
358                     {
359                         if ( strict )
360                         {
361                             throw new XmlPullParserException( "Unrecognised tag: '" + parser.getName() + "'", parser, null);
362                         }
363                     }
364                 }
365             }
366             eventType = parser.next();
367         }
368         return pluginDescriptor;
369     } //-- PluginDescriptor parsePluginDescriptor(String, XmlPullParser, boolean, String)
